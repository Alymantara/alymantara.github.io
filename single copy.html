def mass_function(K_1,K_2,incl,p_orb):
    """
    Function for circular orbits
    """
    return 1./(2.*np.pi*cts.G)*(K_1 + K_2)**2*K_1*p_orb*(24*3600)/cts.M_sun/np.sin(inc_g*np.pi/180.)**3

# Masses for each component
m1_g = mass_function(k2_g,k1_g,inc_g,porb_g)
m2_g = mass_function(k1_g,k2_g,inc_g,porb_g)

# Mass ratio
q_g = m2_g/m1_g

# Orbital Separation
a_g = 1./(2.*np.pi)*porb_g*(24*3600)*(k1_g+k2_g)/np.sin(inc_g*np.pi/180.)/cts.R_sun

#Roche Lobe
rl_g = 0.49*q_g**(2./3.)/(0.6*q_g**(2./3.) + np.log(1.0+q_g**(1./3.)))*a_g
